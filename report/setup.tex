% Custom PDF insertion
\newcommand{\inputpdf}[4]{
    % #1 = title/description
    % #2 = page-nr start
    % #3 = page-nr end
    % #4 = filename/path
    \chapter{#1}\label{pdf:#1}
    \includepdf[width=\textwidth,frame,
        pagecommand={},
        pages={#2-#3}
    ]{#4}
}
